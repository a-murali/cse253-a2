def train_bigram(loader,

note_vocab_size, chord_vocab_size, note_pad=0, chord_aware=False):

V = note_vocab_size C = chord_vocab_size

counts = np.ones((C, V, V)) if chord_aware else np.ones((V, V)) if chord_aware:

counts = np.ones((C, V, V)) else:

counts = np.ones((V, V))

# 2) accumulate counts

for chords_b, melody_b in tqdm(loader, desc="bigram count"): for chord_seq, note_seq in zip(chords_b, melody_b):

prev = note_pad # treat PAD/REST as start symbol for c, n in zip(chord_seq.tolist(), note_seq.tolist()):

counts[c, prev, n] += 1 else:

counts[prev, n] += 1 prev = n

# 3) normalise -> log prob if chord_aware:

probs = counts / counts.sum(axis=-1, keepdims=True)

logprobs = torch.log(torch.tensor(probs, dtype=torch.float32)) return logprobs # shape C Ã— V Ã— V

else:
